API Documentation

Endpoint

GET /text-to-speech?text=<your_text>

Converts text to speech and returns base64-encoded MP3 audio. Max 1000 characters.

Parameters

  • text: Required. The text to convert (max 1000 characters).

Response (JSON)

{
  "status": "success",
  "audio_base64": "<base64_string>",
  "audio_format": "mp3"
}

On error: {"error": "<message>"}

Example Usage

curl "https://tts-api-liart.vercel.app/text-to-speech?text=Hello%20World"

Code Examples

Python

import requests
import base64

url = "https://tts-api-liart.vercel.app/text-to-speech"
params = {"text": "Hello World"}
response = requests.get(url, params=params)
if response.status_code == 200:
    data = response.json()
    if data["status"] == "success":
        with open("audio.mp3", "wb") as f:
            f.write(base64.b64decode(data["audio_base64"]))
        print("Audio downloaded!")
    else:
        print("Error:", data["error"])
else:
    print("HTTP Error:", response.status_code)

JavaScript (Browser)

fetch('/text-to-speech?text=Hello%20World')
  .then(response => response.json())
  .then(data => {
    if (data.status === 'success') {
      const audioBlob = new Blob([atob(data.audio_base64)], { type: 'audio/mp3' });
      const url = URL.createObjectURL(audioBlob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'audio.mp3';
      a.click();
    } else {
      console.error('Error:', data.error);
    }
  })
  .catch(error => console.error('Fetch error:', error));

Node.js

const fs = require('fs');
const https = require('https');

https.get('https://tts-api-liart.vercel.app/text-to-speech?text=Hello%20World', (res) => {
  let data = '';
  res.on('data', (chunk) => { data += chunk; });
  res.on('end', () => {
    const json = JSON.parse(data);
    if (json.status === 'success') {
      fs.writeFileSync('audio.mp3', Buffer.from(json.audio_base64, 'base64'));
      console.log('Audio downloaded!');
    } else {
      console.error('Error:', json.error);
    }
  });
}).on('error', (e) => console.error('HTTPS error:', e));
